def topology(w)
  g = nil
  begin
    g = Topology.beBorn(GraphManager.toGraph(w))
  rescue GraphManagerException => e
    e.printStackTrace()
  rescue WURCSException => e
    e.printStackTrace()
  end
  kotae = GraphManager.toWURCS(g)
  return kotae
end
